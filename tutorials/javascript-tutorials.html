2. DOM Manipulation

Step 2 — Interacting with HTML

The Document Object Model (DOM) lets JavaScript interact with HTML elements. You can change content, styles, and respond to user actions.

JavaScript 
// Select elements
const heading = document.querySelector('.heading');
const button = document.getElementById('myButton');

// Change content
heading.textContent = 'New Heading Text';

// Change styles
heading.style.color = '#00b4d8';
heading.style.fontSize = '2rem';

// Add event listener
button.addEventListener('click', () => {
  heading.classList.toggle('active');
});

// Create new element
const newDiv = document.createElement('div');
newDiv.className = 'card';
newDiv.innerHTML = '<h3>New Card</h3><p>Content here</p>';
document.body.appendChild(newDiv);

3. ES6+ Modern JavaScript

Step 3 — Modern Syntax & Features

ES6 (ECMAScript 2015) introduced powerful features that make JavaScript more concise and readable.

JavaScript 
// Arrow Functions
const add = (a, b) => a + b;
const greet = name => `Hi ${name}!`;

// Destructuring
const person = { name: "Code4Boy", age: 25 };
const { name, age } = person;

const colors = ["red", "green", "blue"];
const [first, second] = colors;

// Spread Operator
const arr1 = [1, 2, 3];
const arr2 = [...arr1, 4, 5];

const obj1 = { a: 1 };
const obj2 = { ...obj1, b: 2 };

// Template Literals
const greeting = `Hello, my name is ${name} and I'm ${age} years old.`;

// Array Methods
const numbers = [1, 2, 3, 4, 5];

const doubled = numbers.map(n => n * 2);
const evens = numbers.filter(n => n % 2 === 0);
const sum = numbers.reduce((acc, n) => acc + n, 0);

// Find method
const found = numbers.find(n => n > 3);

// Some & Every
const hasEven = numbers.some(n => n % 2 === 0);
const allPositive = numbers.every(n => n > 0);

4. Async JavaScript (Promises & Async/Await)

Step 4 — Asynchronous Programming

Async JavaScript lets you perform operations like fetching data from APIs without blocking your code.

JavaScript 
// Callback Pattern (Old)
function fetchData(callback) {
  setTimeout(() => {
    const data = { id: 1, name: "Code4Boy" };
    callback(data);
  }, 1000);
}

fetchData(data => {
  console.log("Callback:", data);
});

// Promise Pattern
function fetchUser() {
  return new Promise((resolve, reject) => {
    setTimeout(() => {
      resolve({ id: 1, name: "Code4Boy" });
    }, 1000);
  });
}

fetchUser()
  .then(user => console.log("Promise:", user))
  .catch(error => console.error(error));

// Async/Await (Modern)
async function getUser() {
  try {
    const user = await fetchUser();
    console.log("Async/Await:", user);
  } catch (error) {
    console.error(error);
  }
}

getUser();

// Fetch API
async function fetchAPI() {
  try {
    const response = await fetch('https://api.github.com/users/code4boy');
    const data = await response.json();
    console.log(data);
  } catch (error) {
    console.error('Error:', error);
  }
}

fetchAPI();

5. Event Handling

Step 5 — User Interactions

Events let your code respond to user actions like clicks, keypresses, mouse movements, and more.

JavaScript 
// Click Event
button.addEventListener('click', (e) => {
  console.log('Clicked!', e.target);
});

// Form Submit
form.addEventListener('submit', (e) => {
  e.preventDefault();
  const formData = new FormData(form);
  console.log(formData.get('email'));
});

// Keyboard Events
input.addEventListener('keydown', (e) => {
  console.log('Key pressed:', e.key);
});

input.addEventListener('keyup', (e) => {
  if (e.key === 'Enter') {
    console.log('Enter pressed!');
  }
});

// Mouse Events
element.addEventListener('mouseenter', () => {
  element.style.background = '#00b4d8';
});

element.addEventListener('mouseleave', () => {
  element.style.background = '';
});

// Input Event (Real-time)
searchInput.addEventListener('input', (e) => {
  const searchTerm = e.target.value.toLowerCase();
  filterItems(searchTerm);
});

// Event Bubbling & Delegation
document.querySelector('.list').addEventListener('click', (e) => {
  if (e.target.matches('.item')) {
    console.log('Item clicked:', e.target.textContent);
  }
});

6. LocalStorage & SessionStorage

Step 6 — Browser Storage

LocalStorage and SessionStorage let you store data in the browser. LocalStorage persists even after closing the browser.

JavaScript 
// LocalStorage - Persistent Storage

// Save data
localStorage.setItem('username', 'Code4Boy');
localStorage.setItem('age', 25);

// Get data
const username = localStorage.getItem('username');
const age = localStorage.getItem('age');

// Remove data
localStorage.removeItem('username');

// Clear all
localStorage.clear();

// Store Objects (JSON.stringify)
const user = { name: 'Code4Boy', skills: ['JS', 'Python'] };
localStorage.setItem('user', JSON.stringify(user));

const storedUser = JSON.parse(localStorage.getItem('user'));

// SessionStorage - Temporary (cleared when tab closes)
sessionStorage.setItem('tempData', 'Some data');
const temp = sessionStorage.getItem('tempData');

// Real Example: Dark Mode Toggle
const themeToggle = document.querySelector('#theme-toggle');
const currentTheme = localStorage.getItem('theme');

if (currentTheme === 'dark') {
  document.body.classList.add('dark-theme');
}

themeToggle.addEventListener('click', () => {
  document.body.classList.toggle('dark-theme');
  const theme = document.body.classList.contains('dark-theme') ? 'dark' : 'light';
  localStorage.setItem('theme', theme);
});
